def answer_one():
    from sklearn.linear_model import LinearRegression
    from sklearn.preprocessing import PolynomialFeatures

    test = np.linspace(0,10,100).reshape(100,1)
    
    result = np.zeros((4,100))
    
    for i, degree in enumerate([1,3,6,9]):
        poly = PolynomialFeatures(degree=degree)
        x_poly = poly.fit_transform(X_train.reshape(11,1))
        linreg = LinearRegression().fit(x_poly, y_train)
        y = linreg.predict(poly.fit_transform(test))
        result[i,:] = y
    
    return result

Question 2¶

Write a function that fits a polynomial LinearRegression model on the training data X_train for degrees 0 through 9. For each model compute the $R^2$ (coefficient of determination) regression score on the training data as well as the the test data, and return both of these arrays in a tuple.

This function should return one tuple of numpy arrays (r2_train, r2_test). Both arrays should have shape (10,)

def answer_two():
    from sklearn.linear_model import LinearRegression
    from sklearn.preprocessing import PolynomialFeatures
    from sklearn.metrics.regression import r2_score

    r2_train = np.zeros(10)
    r2_test = np.zeros(10)
    
    for i in range(10):
        poly = PolynomialFeatures(degree=i)
        
        # train:
        x_train_poly = poly.fit_transform(X_train.reshape(11,1))
        linreg = LinearRegression().fit(x_train_poly, y_train)
        r2_train[i] = linreg.score(x_train_poly, y_train)
        
        # test
        x_test_poly = poly.fit_transform(X_test.reshape(4,1))
        r2_test[i] = linreg.score(x_test_poly, y_test)

    return (r2_train, r2_test)

def answer_six():
    from sklearn.svm import SVC
    from sklearn.model_selection import validation_curve
    
    gamma = np.logspace(-4,1,6)
    this_C = 1
    clf1 = SVC(kernel = 'rbf', C=this_C, random_state=0)
    train_score, test_score = validation_curve(clf1, X_subset, y_subset, param_name = 'gamma', param_range=gamma, 
                                              scoring='accuracy')
    
    train =[]
    test=[]
    for i in train_score:
        train.append(i.mean())
    for j in test_score:
        test.append(j.mean())
    
    scores = (np.array(train), np.array(test))
        
        

    return scores
